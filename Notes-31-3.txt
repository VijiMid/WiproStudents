using System;

// In c sharp all keywords will be in blue color.
// What is Keywords? Its a reserved words. Its reserved for particular operations/ task


namespace CSharpSamples1 // namespace is the keyword . It contains foldername
{
    class Program // class program is the default class in a project
    {
        // A program should contain only one main.
        static void Main(string[] args) // This is line similar to main() of your c lang.
        {
            /*
            int a=10; 
            int b = 20;
            int c;// decalaration Statement
            c = a + b; // processing statement

            Console.WriteLine("Answer is : " + c); 
            // The above line similar to printf - OUtput statement
            // Console-predefined class and WriteLine - predefined method
            */
            // Console.ReadLine() - similar to Scanf of your C lang
            // default readline type is string 
            // What is string? More than one characters/collection of charaters are called a string
            //******************************************************

            //  Convert - predefined class.
            //ToInt32 - to convert  a string to int (type casting )
            Console.WriteLine("Enter a : ");
            int a = Convert.ToInt32( Console.ReadLine());
            Console.WriteLine("Enter b : ");
            int b = Convert.ToInt32(Console.ReadLine());
            int c;
            c = a + b;
            Console.WriteLine("Answer is : " + c);
        }

    }
}

// predefined and user defined classes will be in light blue clolor.
// Program and Console are predefined classes
